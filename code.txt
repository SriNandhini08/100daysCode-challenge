\\prime within a range

#include <stdio.h>
#include <stdbool.h>

bool isprime(int n) {
    if (n < 2) return false; // 0 and 1 are not prime
    for (int i = 2; i < n; i++) {
        if (n % i == 0)
            return false;
    }
    return true;
}

int main() {
    int start, end, i;
    printf("Start: ");
    scanf("%d", &start);
    printf("End: ");
    scanf("%d", &end);

    for (i = start; i <= end; i++) {
        if (isprime(i))
            printf("%d ", i);
    }

    return 0;
}
